```python
def calculate_progressive_stop(position, time_in_trade_minutes, base_stop_pct):
    """
    Tighten stop progressively over time

    Args:
        position: Position dict with entry_price, side, etc.
        time_in_trade_minutes: Minutes since entry
        base_stop_pct: ATR-based initial stop width (e.g., 0.018 for 1.8%)

    Returns:
        Updated stop price
    """
    entry_price = position['entry_price']
    side = position['side']

    # Define tightening schedule
    if time_in_trade_minutes < 5:
        # 0-5 min: Full ATR stop (breathing room)
        stop_width = base_stop_pct
    elif time_in_trade_minutes < 10:
        # 5-10 min: 75% of ATR stop
        stop_width = base_stop_pct * 0.75
    elif time_in_trade_minutes < 15:
        # 10-15 min: 50% of ATR stop
        stop_width = base_stop_pct * 0.50
    else:
        # 15+ min: Move to breakeven if MFE > 0.5%
        # Check if trade has shown 0.5%+ profit at any point
        if position.get('best_mfe_pct', 0) > 0.5:
            return entry_price  # Breakeven
        else:
            # Keep 50% ATR stop if no profit shown
            stop_width = base_stop_pct * 0.50

    # Calculate stop price
    if side == 'LONG':
        stop_price = entry_price * (1 - stop_width)
    else:  # SHORT
        stop_price = entry_price * (1 + stop_width)

    return stop_price
```

def enhanced_fifteen_minute_rule(position, current_price, time_in_trade):
    entry_price = position['entry_price']
    stop_price = position['stop']

    # Calculate current metrics
    if position['side'] == 'LONG':
        gain = current_price - entry_price
        loss = entry_price - current_price
        stop_distance = entry_price - stop_price
    else:
        gain = entry_price - current_price
        loss = current_price - entry_price
        stop_distance = stop_price - entry_price

    gain_pct = (gain / entry_price) * 100
    loss_pct = (loss / entry_price) * 100

    # Rule 1: Original 15-min rule
    if time_in_trade >= 15 and gain_pct < 0.10:
        return True, "15MIN_RULE"

    # Rule 2: Loss exceeds 50% of stop distance
    if loss > (stop_distance * 0.5):

    # Rule 3: Poor MFE/MAE ratio by 10 minutes
    if time_in_trade >= 10:
        mfe = position.get('best_mfe', 0)
        mae = position.get('worst_mae', 0)
        if mae > 0 and (mfe / mae) < 0.3:
            return True, "POOR_MFE_MAE"

    return False, None
